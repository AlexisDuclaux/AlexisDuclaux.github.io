You just started a new job at a bodega in outer space! All your customers are different, and they all speak different languages. Learn to communicate with them to complete their orders by typing the letters you think each symbol corresponds to.

All you need is a mouse and keyboard!

The coloured text represents what the customer wants from the store.
 - Click the items to add them to their cart, and then checkout to complete the transaction. Watch for their reaction to see if you got it right!
 - Click the book icon to access your language handbook and click the arrows to find the right language.
 - Using the customer’s reaction, figure out what they were asking for in order to learn your first batch of letters!
 - Type in letters for each symbol and watch as the dialogue boxes convert to English!

Play right in your web browser.

    public void NewCustomer()

      int u = Custsomers[NewCust].GetComponent().Type;
      Canvas.GetComponent().AlienType = u;
      nbItem = Random.Range(1, 3);
      LastToLast = LastCustomer;
      LastCustomer = NewCust;

      switch (nbItem)
      {
        case 1:
            string[] strg = this.GetComponent().OneObject;
            string[] objs = this.GetComponent().SellObjects;
            string strObj = strg[Random.Range(0, strg.Length)];
            FirstObj = objs[Random.Range(0, objs.Length)];
            string[] Split = strObj.Split(char.Parse("$"));
            NewText = Split[0] + parseColor + FirstObj + parseColorend + Split[1];
            break;

        case 2:
            string[] strg2 = this.GetComponent().TwoObject;
            string[] objs2 = this.GetComponent().SellObjects;
            string strObj2 = strg2[Random.Range(0, strg2.Length)];
            FirstObj = objs2[Random.Range(0, objs2.Length)];
            SecondObj = objs2[Random.Range(0, objs2.Length)];
            while (SecondObj == FirstObj)
            {
              SecondObj = objs2[Random.Range(0, objs2.Length)];
            }
            string[] Split2 = strObj2.Split(char.Parse("$"));
            NewText = Split2[0] + parseColor + FirstObj + parseColorend + Split2[1] + parseColor + SecondObj + parseColorend + Split2[2] ;
            break;

        case 3:
            string[] strg3 = this.GetComponent().ThreeObject;
            string[] objs3 = this.GetComponent().SellObjects;
            string strObj3 = strg3[Random.Range(0, strg3.Length)];
            FirstObj = objs3[Random.Range(0, objs3.Length)];
            SecondObj = objs3[Random.Range(0, objs3.Length)];

            {
              SecondObj = objs3[Random.Range(0, objs3.Length)];
            }
            ThirdObj = objs3[Random.Range(0, objs3.Length)];
            while ((ThirdObj == FirstObj) || (ThirdObj == SecondObj))

              ThirdObj = objs3[Random.Range(0, objs3.Length)];

//Script to spawn and manage when a new client is coming

using System.Collections;
using System.Collections.Generic;
using UnityEngine;

public class MoveSpaceship : MonoBehaviour
{
    public Transform MovePoint;
    public GameObject[] spawns;
    public float speed = 1.0f;
    public bool stop = false;
    public bool depart = false;

    // Start is called before the first frame update
    void Start()
    {
      MovePoint = GameObject.Find("GameManager").GetComponent().Port;
    }

    void Update()
    {
      // Move our position a step closer to the target.
      float step =  speed * Time.deltaTime; // calculate distance to move
      transform.position = Vector3.MoveTowards(transform.position, MovePoint.position, step);
      Vector3 relativePos = MovePoint.position - transform.position;

      // the second argument, upwards, defaults to Vector3.up
      Quaternion rotation = Quaternion.LookRotation(relativePos, Vector3.up);
      transform.rotation = rotation;

      // Check if the position of the cube and sphere are approximately equal.
      if (Vector3.Distance(transform.position, MovePoint.position) < 0.001f)
      {
        if (!stop)
        {
          GameObject.Find("GameManager").GetComponent().NewCustomer();
          stop = true;
        }
        else if (depart)
        {
            GameObject.Find("GameManager").GetComponent().Timer();
            Destroy(this);
        }
      }
    }

    public void Depart()
    {
      spawns = GameObject.Find("GameManager").GetComponent().Spawns;
      MovePoint = spawns[Random.Range(0, spawns.Length)].transform;
      this.GetComponent().Play(0);
      depart = true;
    }
}
                      
                    
                      
//Script that serves as a cypher to decode each alien's dialect

using System.Collections;
using System.Collections.Generic;
using UnityEngine;
using TMPro;

public class Dialect : MonoBehaviour
{
    public GameObject[] Language;
    public int Type;
    public int MaxTypes;

    Transform LangText;
    GameObject Answer;

    Alien Al;
    GameManager Manager;

    // Start is called before the first frame update
    public void Refresh()
    {
      switch (Type)
        {
          case 1:
              Al = GameObject.Find("Imug").GetComponent();
              break;

          case 2:
              Al = GameObject.Find("Xsiale").GetComponent();
              break;

          case 3:
              Al = GameObject.Find("Tirbint").GetComponent();
              break;

          case 4:
              Al = GameObject.Find("Imek").GetComponent();
              break;

          case 5:
              Al = GameObject.Find("Z-XWY").GetComponent();
              break;
        }
      Al.Refresh();
      Manager = GameObject.Find("GameManager").GetComponent();
      for(int i = 0; i < Language.Length; i++)
      {
        LangText = Language[i].transform.Find("Name");
        Answer = Language[i];
        LangText.GetComponent().text = Al.Language + Al.Alph[i];
        Answer.GetComponent().text = Al.Cypher[i];
      }
    }

    // Update is called once per frame
    public void Change()
    {
      Manager = GameObject.Find("GameManager").GetComponent();
      for(int i = 0; i < Language.Length; i++)
      {
        Answer = Language[i];
        switch (Type)
          {
          case 1:
              Manager.Alien_cypher[i] = Answer.GetComponent().text.ToUpper();
              GameObject.Find("Imug").GetComponent().Refresh();
              break;

          case 2:
              Manager.Runes_cypher[i] = Answer.GetComponent().text.ToUpper();
              GameObject.Find("Xsiale").GetComponent().Refresh();
              break;

          case 3:
              Manager.Elvish_cypher[i] = Answer.GetComponent().text.ToUpper();
              GameObject.Find("Tirbint").GetComponent().Refresh();
              break;

          case 4:
              Manager.Mason_cypher[i] = Answer.GetComponent().text.ToUpper();
              GameObject.Find("Imek").GetComponent().Refresh();
              break;

          case 5:
              Manager.ASL_cypher[i] = Answer.GetComponent().text.ToUpper();
              GameObject.Find("Z-XWY").GetComponent().Refresh();
              break;
          }
          //GameObject.Find("Canvas").GetComponent().Refresh();
      }
    }

    public void Up(){
      if(Type < MaxTypes){
        Type ++;
      }
      else{
        Type = 1;
      }
      Refresh();
    }

    public void Down(){
      if(Type > 1){
        Type --;
      }
      else{
        Type = MaxTypes;
      }
      Refresh();
    }
}
