Running With Scissors is a local multiplayer game for up to 4 players! (For the best experience, please use Xbox controllers.)

Your hair constantly grows!

Your objective is to run around with scissors trying to cut the hair of everyone else, all while avoiding the same fate.
The closer you are to cutting someone’s hair at the root, the more lengths of hair you earn. The player with the most hair snipped when the timer runs out, wins!

//Script used to manage the different states of the game

using System;
using System.Collections;
using System.Collections.Generic;
using UnityEngine;
using UnityEngine.SceneManagement;
using UnityEngine.UI;

public class GameManager : MonoBehaviour
{
  public bool gamePaused = false;

  [Header("Players")]
  public List Players = new List();
  public List AvailableColors = new List();
  public List PlayerColors = new List();
  public List AvailableMaterials = new List();
  public List PlayerMaterials = new List();
  public List PlayerScores = new List();

  [Header("Level Info")]
  public int level = 0;
  public List Timers = new List();
  int timerSelect = 2;
  public float choiceTimerLevel = 11.0f;
  public float timerLevel = 11.0f; //3minutes

  void Awake()
  {
      choiceTimerLevel = Timers[timerSelect];
      GameObject[] objs = GameObject.FindGameObjectsWithTag("GameManager");

      if (objs.Length > 1)
      {
          Destroy(this.gameObject);
      }

      DontDestroyOnLoad(this.gameObject);
  }

  public void timerPlus()
  {
      if(timerSelect < Timers.Count -1)
      {
          timerSelect += 1;
      }
      else{
          timerSelect = 0;
      }

      choiceTimerLevel = Timers[timerSelect];
  }

  public void timerLess()
  {
      if(timerSelect > 0)
      {
          timerSelect -= 1;
      }
      else{
          timerSelect = Timers.Count -1;
      }

      choiceTimerLevel = Timers[timerSelect];
  }

  public void startLevel()
  {
      timerLevel = choiceTimerLevel;
      for(int idx = 0; idx < Players.Count; idx++)
      {
          Players[idx].transform.parent = this.gameObject.transform;
          Players[idx].GetComponentInChildren().TrimHair(0);
      }
      SceneManager.LoadScene("Map1", LoadSceneMode.Single);
  }

  void FixedUpdate()
  {
      if(level == 1)
      {
          if(!gamePaused){
              timerLevel -= Time.deltaTime;
              for(int idx = 0; idx < Players.Count; idx++)
              {
                  int feets = Players[idx].transform.GetChild(1).GetComponent().TotalStrandCut;
                  PlayerScores[idx] = feets;
              }
              if(timerLevel <= 0.0f)
              {
                  level = 2;
              }
          }
      }
  }
}
                      
                    
                    
//Character controller

using System.Collections;
using System.Collections.Generic;
using UnityEngine;
using UnityEngine.InputSystem;

public class TopDownController : MonoBehaviour
{

  [SerializeField] CharacterController controller;
  [SerializeField] float Speed = 10.0f;
  [SerializeField] float dashMultiplier = 2f;
  [SerializeField] float dashMaxTimer = 3f;
  [SerializeField] float cutCooldown = 1f;
  [SerializeField] GameObject Model;
  [SerializeField] Animator Anim;
  [SerializeField] GetTargetHair targetedStrands;
  [SerializeField] HairPhysics hairRoot;
  [SerializeField] bool losePointsWhenCuttingOwnHair = true;
  [SerializeField] public int TotalStrandCut = 0;
  //public GameObject particlesContainer;
  public GameObject floatingScore;
  public Transform topHead;

  Vector2 movementInput;
  Vector2 lastMovementInput;
  GameManager manager;
  Color newColor;
  Material newMat;


  private bool canMove = true;
  private float dashTimer = 0f;
  private float cutTimer = 0f;
  private SoundManager sm;
  private GameObject mainCam;
  private float prevSpeed;

  void Awake()
  {
      manager = GameObject.FindGameObjectWithTag("GameManager").GetComponent();
      manager.Players.Add(this.gameObject.transform.parent.gameObject);
      GameObject soundManagerObj = GameObject.FindGameObjectWithTag("SoundManager");
      if(soundManagerObj != null)
          sm = soundManagerObj.GetComponent();
      newColor = manager.AvailableColors[0];
      newMat = manager.AvailableMaterials[0];
      manager.AvailableColors.RemoveAt(0);
      manager.AvailableMaterials.RemoveAt(0);
      manager.PlayerColors.Add(newColor);
      manager.PlayerMaterials.Add(newMat);
      ChangeHairColor();
  }

  void Start(){
      mainCam = GameObject.Find("CameraContainer");
  }

  void ChangeHairColor()
  {
      this.transform.parent.GetChild(0).GetChild(1).gameObject.GetComponent().material = newMat;
  }

  void Update()
  {
      if(mainCam == null)
          mainCam = GameObject.Find("CameraContainer");

      if (manager.level == 1 && !manager.gamePaused)
      {
          if(canMove){
              Vector3 horizontalVelocity = (Vector3.right * -movementInput.y + Vector3.forward * movementInput.x) * Speed;
              controller.Move(horizontalVelocity * Time.deltaTime);
              var rotation = Vector2.SignedAngle(new Vector2(1,0), new Vector2(movementInput.x,-movementInput.y));

              if(movementInput != new Vector2(0,0))
              {
                  Anim.SetFloat("Blend", 1.0f);
                  Model.transform.eulerAngles = new Vector3(0,rotation,0);
              }
              else
              {
                  Anim.SetFloat("Blend", 0.0f);
              }
              lastMovementInput = movementInput;
          }else{
              Vector3 horizontalVelocity = (Vector3.right * -lastMovementInput.y + Vector3.forward * lastMovementInput.x) * (Speed * dashMultiplier);
              controller.Move(horizontalVelocity * Time.deltaTime);
              dashTimer -= Time.deltaTime;
              if(dashTimer <= 0){
                  canMove = true;
              }
          }
      }
      if(cutTimer > 0){
          cutTimer -= Time.deltaTime;
      }
  }

  public void OnMove(InputAction.CallbackContext ctx) => movementInput = ctx.ReadValue();

  public void OnStartPressed(InputAction.CallbackContext ctx)
  {
      if (manager.level == 0 && manager.Players.Count >= 2)
      {
          GameObject.FindGameObjectWithTag("StartMenu").GetComponent().Transition();
      }else if(manager.level == 1){
          if(!manager.gamePaused){
              prevSpeed = Anim.speed;
              Anim.speed = 0;
              manager.gamePaused = true;
              Time.timeScale = 0;
          }else{
              Anim.speed = prevSpeed;
              manager.gamePaused = false;
              Time.timeScale = 1;
          }
      }
  }

  public void OnTimerPlusPressed(InputAction.CallbackContext ctx)
  {
      if (manager.level == 0 && ctx.started)
      {
          manager.timerPlus();
      }
  }

  public void OnTimerLessPressed(InputAction.CallbackContext ctx)
  {
      if (manager.level == 0 && ctx.started)
      {
          manager.timerLess();
      }
  }

  public void OnCutPressed(InputAction.CallbackContext ctx)
  {
      if (manager.level == 1 && !manager.gamePaused)
      {
          if(cutTimer <= 0){
              if(lastMovementInput != new Vector2(0,0)){
                  dashTimer = dashMaxTimer;
                  if(sm != null)
                      sm.PlayAudio("DashSFX");

                  canMove = false;
              }
              var strands = targetedStrands.GetHairStrands();
              int totalStrands = 0;
              foreach(var strand in strands){
                  if(strand != null){
                      var checkStrand = strand.GetComponent();
                      var root = checkStrand.GetHairRoot();
                      int newStrand = 0;
                      if(root != hairRoot){
                          newStrand = root.CutFromStrand(strand);
                      }
                      totalStrands += newStrand;
                  }
              }
              if(totalStrands > 0){
                  if(sm != null)
                      sm.PlayAudio("NormalCut");

                  if(mainCam != null){
                      mainCam.GetComponent().ShakeCamera();
                  }
                  TotalStrandCut += totalStrands;
                  if(floatingScore != null){
                      var score = Instantiate(floatingScore);
                      score.transform.SetParent(this.gameObject.transform.parent);
                      score.gameObject.transform.position = topHead.position;
                      score.GetComponent().SetInstance(totalStrands,newColor,topHead);
                  }
              }
              cutTimer = cutCooldown;
          }
      }
  }
}
                    
                  
                    
using System.Collections;
using System.Collections.Generic;
using UnityEngine;
using UnityEngine.UI;
using UnityEngine.SceneManagement;
using TMPro;

public class EndMenu : MonoBehaviour
{
  GameManager manager;
  private SoundManager sm;
  public List PlayerScores = new List();
  public List AvailableColors = new List();
  public List AvailableMaterials = new List();
  public int biggestScore;
  public int strongestPlayer;
  public float fillSpeed = 10.0f;

  int state = -1;

  void Start(){
      GameObject soundManagerObj = GameObject.FindGameObjectWithTag("SoundManager");
      if(soundManagerObj != null)
          sm = soundManagerObj.GetComponent();

      if(sm != null)
          sm.PlayAudio("MainMenuMusic");

      manager = GameObject.FindGameObjectWithTag("GameManager").GetComponent();
      for(int idx = 0; idx < manager.Players.Count; idx++)
      {
          manager.Players[idx].gameObject.SetActive(false);
          this.transform.GetChild(3).GetChild(idx).gameObject.SetActive(true);
          this.transform.GetChild(3).GetChild(idx).gameObject.GetComponent().color = manager.PlayerColors[idx];
          var scoreInt = (int)PlayerScores[idx];
          this.transform.GetChild(3).GetChild(idx).GetChild(0).gameObject.GetComponent().text = scoreInt.ToString() + " ft";
      }

      List ScoresAmount = new List();
      for(int u = 0; u < manager.PlayerScores.Count; u++)
      {
          ScoresAmount.Add(manager.PlayerScores[u]);
      }
      ScoresAmount.Sort();
      biggestScore = ScoresAmount[ScoresAmount.Count-1];
  }

  void FixedUpdate()
  {
      if (state == 0)
      {
          for(int idx = 0; idx < manager.Players.Count; idx++)
          {
              this.transform.GetChild(3).GetChild(idx).gameObject.GetComponent().color = manager.PlayerColors[idx];
              if(PlayerScores[idx] < manager.PlayerScores[idx])
              {
                  if (PlayerScores[idx] + Time.deltaTime * fillSpeed <= manager.PlayerScores[idx])
                  {
                      PlayerScores[idx] += Time.deltaTime * fillSpeed;
                  }
                  else
                  {
                      PlayerScores[idx] = manager.PlayerScores[idx];
                  }

              }
              var scoreInt = (int)PlayerScores[idx];
              this.transform.GetChild(3).GetChild(idx).GetChild(0).gameObject.GetComponent().text = scoreInt.ToString() + " ft";
              var fillAmounts = scoreInt/1000.0f;
              this.transform.GetChild(3).GetChild(idx).gameObject.GetComponent().fillAmount = fillAmounts;
              if(scoreInt >= biggestScore)
              {
                  strongestPlayer = idx;
                  state = 1;
              }
          }
      }
      else if (state == 1)
      {
          this.transform.GetChild(3).GetChild(4).gameObject.SetActive(true);
          int winner = strongestPlayer + 1;
          this.transform.GetChild(3).GetChild(4).GetChild(0).gameObject.GetComponent().text = winner.ToString();
          this.transform.GetChild(0).gameObject.SetActive(true);
          this.transform.GetChild(1).gameObject.SetActive(true);
      }
  }

  public void onMenuLoaded()
  {
      state = 0;
  }

  public void OnMainMenu()
  {
      GameManager manager = GameObject.FindGameObjectWithTag("GameManager").GetComponent();
      for(int idx = 0; idx < manager.Players.Count; idx++)
      {
          Destroy(manager.Players[idx].gameObject);
      }

      int numPlayers = manager.Players.Count;

      for(int idx = 0; idx < numPlayers; idx++)
      {
          manager.Players.RemoveAt(0);
      }

      manager.AvailableColors = new List();
      manager.PlayerColors = new List();
      manager.AvailableMaterials = new List();
      manager.PlayerMaterials = new List();

      for(int idx = 0; idx < AvailableColors.Count; idx++)
      {
          manager.AvailableColors.Add(AvailableColors[idx]);
      }

      for(int idx = 0; idx < AvailableMaterials.Count; idx++)
      {
          manager.AvailableMaterials.Add(AvailableMaterials[idx]);
      }

      for(int idx = 0; idx < manager.PlayerScores.Count; idx++)
      {
          manager.PlayerScores[idx] = 0;
      }

      manager.level = 0;
      SceneManager.LoadScene("MainMenu", LoadSceneMode.Single);
  }

  public void OnPushedRestart()
  {
      GetComponent().Play("EndMenuRestart");
  }

  public void OnRestart()
  {
      GameManager manager = GameObject.FindGameObjectWithTag("GameManager").GetComponent();

      for(int idx = 0; idx < manager.PlayerScores.Count; idx++)
      {
          manager.PlayerScores[idx] = 0;
      }

      for(int idx = 0; idx < manager.Players.Count; idx++)
      {
          manager.Players[idx].GetComponentInChildren().enableHairGrowth = false;
          manager.Players[idx].GetComponentInChildren().TrimHair(0);
          manager.Players[idx].GetComponentInChildren().TotalStrandCut = 0;
      }

      manager.timerLevel = manager.choiceTimerLevel;
      manager.level = 0;
      SceneManager.LoadScene("Map1", LoadSceneMode.Single);
  }
}
