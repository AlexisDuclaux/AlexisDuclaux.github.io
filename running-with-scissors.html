Running With Scissors is a local multiplayer game for up to 4 players! (For the best experience, please use Xbox controllers)

Your hair constantly grows.

Your objective is to run around with scissors trying to cut the hair of everyone else, all while avoiding the same fate.

The closer you are to cutting someone’s hair at the root, the more lengths of hair you earn. The player with the most hair snipped when the timer runs out, wins!

CitizenManager.cs


  using System.Collections;
  using System.Collections.Generic;
  using UnityEngine;

  [System.Serializable]
  public class Citizen
  {
      public int address;
      public int office;
      public int entertainment;

      public int spawn;
      public int desk;
      public int table;

      public int backtoHome;
      public int gotoWork;
      public int gotoEntertain;
  }

  [System.Serializable]
  public class CitizensList
  {
      public Citizen[] Citizens;
  }

  public class CitizenManager : MonoBehaviour
  {
      [SerializeField] private GameObject Appartments;
      [SerializeField] private GameObject Offices;
      [SerializeField] private GameObject Bars;

      List aGrid = new List();
      List oGrid = new List();
      List bGrid = new List();

      List aTransform = new List();
      List oTransform = new List();
      List bTransform = new List();

      List aTime = new List();
      List oTime = new List();
      List bTime = new List();

      List toTarget = new List();

      public TextAsset jsonFile;

      private bool hasAllSpawned = false;

      void Start()
      {
          CitizensList citizensInJson = JsonUtility.FromJson(jsonFile.text);
          foreach (Citizen citizen in citizensInJson.Citizens)
          {
              aGrid.Add(citizen.address);
              oGrid.Add(citizen.office);
              bGrid.Add(citizen.entertainment);

              aTransform.Add(citizen.spawn);
              oTransform.Add(citizen.desk);
              bTransform.Add(citizen.table);

              aTime.Add(citizen.backtoHome);
              oTime.Add(citizen.gotoWork);
              bTime.Add(citizen.gotoEntertain);
          }

          for(int i = 0; i < transform.childCount; i++)
          {
              if((GetComponent().hour > aTime[i] && GetComponent().hour < oTime[i]) || ((oTime[i] - aTime[i] < 0) && (GetComponent().hour > aTime[i] || GetComponent().hour < oTime[i])))
              {
                transform.GetChild(i).GetComponent().Warp(Appartments.transform.GetChild(aGrid[i]).GetChild(2).GetChild(aTransform[i]).position);
                toTarget.Add(Appartments.transform.GetChild(aGrid[i]).GetChild(2).GetChild(aTransform[i]));
              }

              else if((GetComponent().hour > oTime[i] && GetComponent().hour < bTime[i]) || ((bTime[i] - oTime[i] < 0) && (GetComponent().hour > oTime[i] || GetComponent().hour < bTime[i])))
              {
                transform.GetChild(i).GetComponent().Warp(Offices.transform.GetChild(oGrid[i]).GetChild(2).GetChild(oTransform[i]).position);
                toTarget.Add(Offices.transform.GetChild(oGrid[i]).GetChild(2).GetChild(oTransform[i]));
              }

              else
              {
                transform.GetChild(i).GetComponent().Warp(Bars.transform.GetChild(bGrid[i]).GetChild(2).GetChild(bTransform[i]).position);
                toTarget.Add(Bars.transform.GetChild(bGrid[i]).GetChild(2).GetChild(bTransform[i]));
              }
          }
          hasAllSpawned = true;
      }

      void FixedUpdate()
      {
        if(hasAllSpawned)
        {
          for(int i = 0; i < transform.childCount; i++)
          {
            transform.GetChild(i).GetComponent().destination = toTarget[i].position;
          }
        }
      }

      public void checkPos()
      {
        for(int i = 0; i < transform.childCount; i++)
        {
            if(GetComponent().hour == aTime[i])
            {
              toTarget[i] = Appartments.transform.GetChild(aGrid[i]).GetChild(2).GetChild(aTransform[i]);
            }

            else if(GetComponent().hour == oTime[i])
            {
              toTarget[i] = Offices.transform.GetChild(oGrid[i]).GetChild(2).GetChild(oTransform[i]);
            }

            else if(GetComponent().hour == bTime[i])
            {
              toTarget[i] = Bars.transform.GetChild(bGrid[i]).GetChild(2).GetChild(bTransform[i]);
            }
        }
      }
    }
  }

CameraMove.cs


  using System.Collections;
  using System.Collections.Generic;
  using UnityEngine;

  public class CameraMove : MonoBehaviour
  {
    public float speed = 20;
    private Vector2 motion;

    void Update()
    {
      motion = new Vector2(Input.GetAxisRaw("Horizontal"), Input.GetAxisRaw("Vertical"));
      transform.Translate(motion*speed*Time.deltaTime);
    }

  }
