It’s been far too long since you’ve been able to get some sleep.
Why? Your house is haunted! Ghosts screech, scream, and holler for the items that will finally lay them to rest.
You’re tasked with navigating your home, eyelids heavy, trying to find the objects that will allow them to move on.
Once they do, you’ll finally get some peace and quiet!

How to Play:

There are 7 ghosts to give objects to. Pick an object up and push it into the ghost. Ghosts will fade in and out depending on the time of day.

Keep an eye on your fatigue meter in the lower right-hand corner.
You can manually blink to give yourself more time before you’re forced to go to bed.
Be careful, though! Whenever an object is not in your view, tricky poltergeists will move it somewhere else!

//Script used to manage the day and night cycle

using System.Collections;
using System.Collections.Generic;
using UnityEngine;
using UnityEngine.UI;

public class DayNight : MonoBehaviour
{
    public float hourClock = 0.8f;
    public float secondsinaday;
    public float IntensityDay;
    public float IntensityNight;
    public float timerCurveTime;
    public float rotCurveTime;
    public GameObject DayLight;
    public AnimationCurve timerCurve;
    public AnimationCurve rotCurve;
    public Color Day;
    public Color Night;
    public Image DAYNIGHTUI;

    public int days = 0;

    // Start is called before the first frame update
    void Start()
    {

    }

    // Update is called once per frame
    void Update()
    {
      if (hourClock >= 2.4f)
      {
        hourClock = 0.0f;
        days++;
      }
      else
      {
        hourClock += Time.deltaTime *(2.4f/secondsinaday);
      }

      timerCurveTime = timerCurve.Evaluate(hourClock);
      rotCurveTime = rotCurve.Evaluate(hourClock);
      DayLight.GetComponent().color = Color.Lerp(Day, Night, timerCurveTime);
      float Rotation = Mathf.Lerp(0, 360, rotCurveTime);
      float Intensity = Mathf.Lerp(4.0f, 0.5f, timerCurveTime);
      DayLight.GetComponent().intensity = Intensity;
      DayLight.transform.rotation = Quaternion.Euler(-200f, Rotation, 0);
      float lightInt = Mathf.Lerp(IntensityDay, IntensityNight, timerCurveTime);
      RenderSettings.ambientIntensity = lightInt;
      RenderSettings.ambientLight = Color.Lerp(Day, Night, timerCurveTime);
      float UI = Mathf.Lerp(-135f, -495.0f, rotCurveTime);
      DAYNIGHTUI.GetComponent().transform.rotation = Quaternion.Euler(0, 0, UI);
    }
}
                      
                    
                      
//Script to manage the object pickups

using System.Collections;
using System.Collections.Generic;
using UnityEngine;

public class PickupObject : MonoBehaviour
{
    public float waitOnPickup = 0.2f;
    public float breakForce = 35f;
    public string refPrefabName;
    [HideInInspector] public bool pickedUp = false;
    [HideInInspector] public PlayerInteractions playerInteractions;
    private Material targetMat;
    Renderer renderer;

    void Awake(){
        targetMat = Resources.Load("Material/Outline", typeof(Material)) as Material;
        renderer = GetComponent ();
        Material[] mats = renderer.materials;
        if(targetMat){
            mats = AddtoArray(mats,targetMat);
            renderer.materials = mats;
        }
    }

    private void OnCollisionEnter(Collision collision){
        if(pickedUp)
        {
            if(collision.relativeVelocity.magnitude > breakForce)
            {
                playerInteractions.BreakConnection();
            }

        }
    }

    //this is used to prevent the connection from breaking when you just picked up the object as it sometimes fires a collision with the ground or whatever it is touching
    public IEnumerator PickUp(){
        yield return new WaitForSecondsRealtime(waitOnPickup);
        int layerIndex = LayerMask.NameToLayer("Picked-up");
        Physics.IgnoreLayerCollision(0, layerIndex,false);
        pickedUp = true;
    }

     T[] AddtoArray(T[] Org, T New_Value){
        T[] New = new T[Org.Length + 1];
        Org.CopyTo(New, 0);
        New[Org.Length] = New_Value;
        return New;
    }
}
                      
                    
                      
//Script that manages all changes in the main menu

using System.Collections;
using System.Collections.Generic;
using UnityEngine;
using UnityEngine.UI;

public class MainMenuManager : MonoBehaviour
{
    public GameObject Main;
    public GameObject Options;
    public GameObject Credits;
    public GameObject HowTo;
    public GameObject back;

    public GameObject playbutton;
    public GameObject creditsbutton;
    public GameObject howtobutton;
    public GameObject optionsbutton;

    public GameObject MusicValue;
    public GameObject ScreamValue;
    public GameObject FootstepValue;
    public GameObject FXValue;

    // Start is called before the first frame update
    public void OnOptions()
    {
      Main.SetActive(false);
      Options.SetActive(true);
      Credits.SetActive(false);
      HowTo.SetActive(false);
      back.SetActive(true);
      AudioOptions man = Object.FindObjectOfType();
      MusicValue.GetComponent().value = man.getMusic();
      ScreamValue.GetComponent().value = man.getScream();
      FootstepValue.GetComponent().value = man.getFoot();
      FXValue.GetComponent().value = man.getFX();
      Reset();
    }

    public void OnCredits()
    {
      Main.SetActive(false);
      Options.SetActive(false);
      Credits.SetActive(true);
      HowTo.SetActive(false);
      back.SetActive(true);
      Reset();
    }

    public void OnHowTo()
    {
      Main.SetActive(false);
      Options.SetActive(false);
      Credits.SetActive(false);
      HowTo.SetActive(true);
      back.SetActive(true);
      Reset();
    }

    public void OnMain()
    {
      Main.SetActive(true);
      Options.SetActive(false);
      Credits.SetActive(false);
      HowTo.SetActive(false);
      back.SetActive(false);
      back.GetComponent().transform.localPosition = new Vector3(back.GetComponent().transform.localPosition.x, -155f, back.GetComponent().transform.localPosition.z);
    }

    public void Onquit()
    {
      Application.Quit();
    }

    public void OnPlay()
    {
      Application.LoadLevel("House_Alex3");
    }

    void Reset()
    {
      playbutton.GetComponent().transform.localPosition = new Vector3(playbutton.GetComponent().transform.localPosition.x, -155f, playbutton.GetComponent().transform.localPosition.z);
      creditsbutton.GetComponent().transform.localPosition = new Vector3(creditsbutton.GetComponent().transform.localPosition.x, -155f, creditsbutton.GetComponent().transform.localPosition.z);
      howtobutton.GetComponent().transform.localPosition = new Vector3(howtobutton.GetComponent().transform.localPosition.x, -155f, howtobutton.GetComponent().transform.localPosition.z);
      optionsbutton.GetComponent().transform.localPosition = new Vector3(optionsbutton.GetComponent().transform.localPosition.x, -155f, optionsbutton.GetComponent().transform.localPosition.z);
    }

    public void setMusic()
    {
      AudioOptions man = Object.FindObjectOfType();
      man.setMusic(MusicValue.GetComponent().value);
    }

    public void setScream()
    {
      AudioOptions man = Object.FindObjectOfType();
      man.setScream(ScreamValue.GetComponent().value);
    }

    public void setFoot()
    {
      AudioOptions man = Object.FindObjectOfType();
      man.setFoot(FootstepValue.GetComponent().value);
    }

    public void setFX()
    {
      AudioOptions man = Object.FindObjectOfType();
      man.setFX(FXValue.GetComponent().value);
    }
}
